for文　繰り返し

  • For
    • for(let i = 0; i < 10; i++)
       {
          console.log(i);
        }

  • continue スキップ
    • for(var a = 0; a < 10; a++)
        {
        if(a === 6)
        {
          continue;
        }
        console.log(a);
      }

  • break 止まる
    • for(let y = 0; y < 10; y++)
      {
        if(y === 6)
        {
          break;
        }
        console.log(y);
      }
